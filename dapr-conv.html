using RvGateway.Models;
using TIBCO.Rendezvous;
using YamlDotNet.Serialization;
using YamlDotNet.Serialization.NamingConventions;
using System.Collections.Concurrent;

namespace RvGateway.Services;

public class RvConnectionManager
{
    private readonly string _configPath;
    private readonly ConcurrentDictionary<string, (TibrvTransport, RvConnectionConfig)> _connections = new();

    public RvConnectionManager(string configPath)
    {
        _configPath = configPath;
        Tibrv.Open(Tibrv.IMPL_NATIVE);
    }

    public RvConnectionConfig? GetConfig(string serviceName)
    {
        using var reader = new StreamReader(_configPath);
        var yaml = new DeserializerBuilder()
            .WithNamingConvention(CamelCaseNamingConvention.Instance)
            .Build();

        var configs = yaml.Deserialize<Dictionary<string, RvConnectionConfig>>(reader);
        return configs.TryGetValue(serviceName, out var cfg) ? cfg : null;
    }

    public TibrvTransport GetOrCreateTransport(string serviceName)
    {
        if (_connections.TryGetValue(serviceName, out var entry))
            return entry.Item1;

        var cfg = GetConfig(serviceName)
            ?? throw new Exception($"No config for service '{serviceName}'");

        var transport = new TibrvNetTransport(cfg.Service, cfg.Network, cfg.Daemon);
        _connections[serviceName] = (transport, cfg);
        return transport;
    }
}
